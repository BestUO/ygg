131 template<class InnerNode>
132 class Compare {
133 public:
134  using PointDescription = std::pair<const typename InnerNode::KeyT, const int_fast8_t>;
135 
136  bool operator()(const InnerNode & lhs,
137  const InnerNode & rhs) const
138  {
139 #pragma GCC diagnostic push
140 #pragma GCC diagnostic ignored "-Wfloat-equal"
141  if (lhs.get_point() != rhs.get_point()) {
142 #pragma GCC diagnostic pop
143  return lhs.get_point() < rhs.get_point();
144  } else {
145  /*
146  * At the same point, the order is: open ends, closed starts, closed ends, open starts
147  */
148  int_fast8_t lhs_priority;
149  if (lhs.is_closed()) {
150  if (lhs.is_start()) {
151  lhs_priority = -1;
152  } else {
153  lhs_priority = 1;
154  }
155  } else {
156  if (lhs.is_start()) {
157  lhs_priority = 2;
158  } else {
159  lhs_priority = -2;
160  }
161  }
162 
163  int_fast8_t rhs_priority;
164  if (rhs.is_closed()) {
165  if (rhs.is_start()) {
166  rhs_priority = -1;
167  } else {
168  rhs_priority = 1;
169  }
170  } else {
171  if (rhs.is_start()) {
172  rhs_priority = 2;
173  } else {
174  rhs_priority = -2;
175  }
176  }
177 
178  return lhs_priority < rhs_priority;
179  }
180  }
181 
182  bool operator()(const typename InnerNode::KeyT & lhs, const InnerNode & rhs) const
183  {
184 #pragma GCC diagnostic push
185 #pragma GCC diagnostic ignored "-Wfloat-equal"
186  if (lhs != rhs.get_point()) {
187  return lhs < rhs.get_point();
188  }
189 #pragma GCC diagnostic pop
190 
191  // only open starts are strictly larger than the key
192  return (rhs.is_start() && !rhs.is_closed());
193  }
194 
195  bool operator()(const InnerNode & lhs, const typename InnerNode::KeyT & rhs) const
196  {
197 #pragma GCC diagnostic push
198 #pragma GCC diagnostic ignored "-Wfloat-equal"
199  if (lhs.get_point() != rhs) {
200  return lhs.get_point() < rhs;
201  }
202 #pragma GCC diagnostic pop
203 
204  // only open ends must strictly go before the key
205  return (!lhs.is_start() && !lhs.is_closed());
206  }
207 
208 
209  bool operator()(const PointDescription & lhs, const InnerNode & rhs) const
210  {
211 #pragma GCC diagnostic push
212 #pragma GCC diagnostic ignored "-Wfloat-equal"
213  if (lhs.first != rhs.get_point()) {
214  return lhs.first < rhs.get_point();
215  }
216 #pragma GCC diagnostic pop
217 
218  if (lhs.second > 0) {
219  // the query is left-open, i.e., it should not to before anything
220  return false;
221  } else if (lhs.second < 0) {
222  // the query is right-open. It should go before everything but open end nodes
223  return rhs.is_start() || rhs.is_closed();
224  } else {
225  // The query is closed.
226  // only open starts are strictly larger than the key
227  return (rhs.is_start() && !rhs.is_closed());
228  }
229  }
230 
231  bool operator()(const InnerNode & lhs, const PointDescription & rhs) const
232  {
233 #pragma GCC diagnostic push
234 #pragma GCC diagnostic ignored "-Wfloat-equal"
235  if (lhs.get_point() != rhs.first) {
236  return lhs.get_point() < rhs.first;
237  }
238 #pragma GCC diagnostic pop
239 
240  if (rhs.second > 0) {
241  // the query is left-open, i.e., everything but left-open is before it
242  return !(lhs.is_start() && !lhs.is_closed());
243  } else if (rhs.second > 0) {
244  // the query is right-open, i.e., nothing must ever strictly go before it
245  return false;
246  } else {
247  // the query is closed
248  // only open ends must strictly go before the key
249  return (!lhs.is_start() && !lhs.is_closed());
250  }
251  }
